    def get_id(self, o: int):
        # 'self.index is np.nan' sometimes cause bug: 'TypeError: 'float' object is not subscriptable'
        if np.all(np.isnan(self.index)):
            return o
        else:
            return self.index[o]



    @property
    def N(self):
        return self._N

    @property
    def m(self):
        return self._m

    @property
    def s(self):
        return self._s
